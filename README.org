#+begin_src sql
CREATE MATERIALIZED VIEW IF NOT EXISTS btc_avg_1day
WITH (timescaledb.continuous) AS
SELECT
    time_bucket('1 day', time) AS day,
    avg(open) as average
FROM
    btc_ohlc_5min
GROUP BY
    day
;
#+end_src

***** Moving Averages

200-week moving average using window functions on weekly aggregates:

#+headers: :engine postgresql :cmdline "postgres://postgres:postgres@localhost:5432/postgres"
#+begin_src sql
DROP VIEW btc_weekly_moving_avg;
#+end_src

#+RESULTS:
| DROP MATERIALIZED VIEW |
|------------------------|

#+headers: :engine postgresql :cmdline "postgres://postgres:postgres@localhost:5432/postgres"
#+begin_src sql
CREATE VIEW btc_moving_avg_1week AS
SELECT
        week,
        avg(average) OVER(ORDER BY week ROWS BETWEEN 199 PRECEDING AND CURRENT ROW) as weekly_moving_avg_200
FROM
        btc_avg_1week
;
#+end_src

#+RESULTS:
| CREATE VIEW |
|-------------|

#+headers: :engine postgresql :cmdline "postgres://postgres:postgres@localhost:5432/postgres"
#+begin_src sql

LIMIT 3;
#+end_src
