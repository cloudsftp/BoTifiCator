#+begin_src sql
CREATE MATERIALIZED VIEW IF NOT EXISTS
    btc_avg_1day
WITH (timescaledb.continuous) AS
SELECT
    time_bucket('1 day', time) AS day,
    avg(open) as average
FROM
    btc_ohlc_5min
GROUP BY
    day
;
ALTER MATERIALIZED VIEW
    btc_avg_1day
SET
    (timescaledb.materialized_only = false)
;
#+end_src

#+headers: :engine postgresql :cmdline "postgres://postgres:postgres@localhost:5432/postgres"

#+headers: :engine postgresql :cmdline "postgres://postgres:postgres@localhost:5432/postgres"
#+begin_src sql
CREATE OR REPLACE VIEW
    btc_moving_avg_1week
AS SELECT
    time_bucket('1 week', week) as week,
    avg(average) OVER(ORDER BY week ROWS BETWEEN 199 PRECEDING AND CURRENT ROW) as moving_average_200,
    avg(average) OVER(ORDER BY week ROWS BETWEEN  99 PRECEDING AND CURRENT ROW) as moving_average_100
FROM
    btc_avg_1week
ORDER BY
    week
DESC
;
#+end_src

SELECT
    w.week,
    w.average,
    m.moving_average_200,
    m.moving_average_100
FROM btc_avg_1week w
JOIN btc_moving_avg_1week m
ON
    w.week = m.week
ORDER BY w.week DESC
LIMIT 2
