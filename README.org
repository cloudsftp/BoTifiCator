*** Moving Averages

#+headers: :engine postgresql :cmdline "postgres://postgres:mysecretpassword@localhost:5432/postgres"
#+begin_src sql
-- Verify the data
SELECT
    day,
    avg(average) over (ORDER BY day ROWS BETWEEN 110 PRECEDING AND CURRENT ROW) AS ma111,
    2 * avg(average) over (ORDER BY day ROWS BETWEEN 350 PRECEDING AND CURRENT ROW) AS ma350x2
FROM btc_daily_avg
ORDER BY day DESC
LIMIT 14;
#+end_src
