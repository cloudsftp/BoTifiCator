#+begin_src sql
SELECT
    time,
    close,
    AVG(close) OVER (ORDER BY time ROWS BETWEEN 288 PRECEDING AND CURRENT ROW) AS moving_average
FROM
    btc_5min
ORDER BY
    time;
#+end_src
