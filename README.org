*** Moving Averages

#+headers: :engine postgresql :cmdline "postgres://postgres:mysecretpassword@localhost:5432/postgres"
#+begin_src sql
SELECT
        day,
        average,
        avg(average) over (ORDER BY day DESC ROWS BETWEEN CURRENT ROW AND 110 FOLLOWING) AS ma111,
        2 * avg(average) over (ORDER BY day DESC ROWS BETWEEN CURRENT ROW AND 349 FOLLOWING) AS ma350x2
FROM btc_daily_avg
ORDER BY day
LIMIT 3;
#+end_src
