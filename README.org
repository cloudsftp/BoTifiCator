#+begin_src sql
CREATE MATERIALIZED VIEW IF NOT EXISTS btc_daily_avg
WITH (timescaledb.continuous) AS
SELECT
        time_bucket('1 day', time) AS day,
        avg(open) as daily_avg
FROM
        btc_5min
GROUP BY
        day;
#+end_src

        time_bucket('1 week', day) AS week,

***** Moving Averages

200-week moving average using window functions on weekly aggregates:

#+headers: :engine postgresql :cmdline "postgres://postgres:postgres@localhost:5432/postgres"
#+begin_src sql
DROP VIEW btc_weekly_moving_avg;
#+end_src

#+RESULTS:
| DROP MATERIALIZED VIEW |
|------------------------|

#+headers: :engine postgresql :cmdline "postgres://postgres:postgres@localhost:5432/postgres"
#+begin_src sql
CREATE VIEW IF NOT EXISTS btc_weekly_moving_avg AS
SELECT
        week,
        avg(weekly_avg) OVER(ORDER BY week ROWS BETWEEN 199 PRECEDING AND CURRENT ROW) as weekly_moving_avg_200
FROM
        btc_weekly_avg;
#+end_src

#+RESULTS:
| SELECT 323 |
|------------|

#+headers: :engine postgresql :cmdline "postgres://postgres:postgres@localhost:5432/postgres"

ORDER BY week DESC
LIMIT 3;
#+end_src
